% Quiz 5: Task

\section{Task}

\frame{
\frametitle{Quiz 5}
\begin{Large}\noindent \textbf{Prove the following equation for the Standard Normal Distribution:}
\begin{equation*}
f\left(x\right)=\frac{1}{2\pi}\cdot\int_{-\infty}^{\infty} \exp\left(-\mathbf{i}tx\right)\cdot\varphi_X(t)\ dt \eqno  (\ast)
\end{equation*}
\end{Large}
}

% Quiz 5: Proof
\section{Proof}

\frame{
\frametitle{Proof}
Let $X\sim N(0,1)$. Define
\begin{equation*}
G\left(x\right)\stackrel{\operatorname{def}}{=}\frac{1}{2\pi}\cdot \int_{-\infty}^{\infty} \exp\left(-\mathbf{i}tx\right)\cdot \varphi_X\left(t\right)\ dt
\end{equation*}
In Quiz 4, it was shown that
\begin{equation*}
\varphi_X\left(t\right) \stackrel{\operatorname{def}}{=}\operatorname{E}\left[\exp\left(\mathbf{i}tX\right)\right] 
											  =\exp\left(-\frac{t^2}{2}\right) \hspace{12mm}  (1)
\end{equation*}
with $t\in\mathbbm{R}$.

}

\frame{
\frametitle{Proof}
Hence, we get
\begin{equation*}
\begin{align*}
 G\left(x\right)&\stackrel{\operatorname{(1)}}{=}\frac{1}{2\pi}\cdot \int_{-\infty}^{\infty} \exp\left(-\mathbf{i}tx\right)\cdot \exp\left(-\frac{t^2}{2}\right)\  dt \\
								&=\left\{\frac{1}{2\pi}\cdot \int_{-\infty}^{\infty} \exp\left(-\mathbf{i}tx\right)\cdot \exp\left(-\frac{t^2}{2}\right)\  dt\right\}\cdot \frac{\sqrt{2\pi}}{\sqrt{2\pi}} \\
								&= \frac{\sqrt{2\pi}}{2\pi} \int_{-\infty}^{\infty} \exp\left(-\mathbf{i}tx\right)\frac{1}{\sqrt{2\pi}}\cdot \exp\left(-\frac{t^2}{2}\right)\  dt \\
\end{align*}
\end{equation*}
}

\frame{
\frametitle{Proof}
Let $f_X$ be the probability density function of the Standard Normal Distribution. Then:
\begin{equation*}
\begin{align*}
G\left(x\right)&= \frac{1}{\sqrt{2\pi}}\cdot \int_{-\infty}^{\infty} \exp\left(-\mathbf{i}tx\right)\cdot f_X\left(t\right)\  dt \\
									&= \frac{1}{\sqrt{2\pi}}\cdot \operatorname{E}\left[\exp\left(\mathbf{i}xX\right)\right] \\
									&\stackrel{\operatorname{(1)}}{=} \frac{1}{\sqrt{2\pi}}\cdot\exp\left(-\frac{x^2}{2}\right)
\end{align*}
\end{equation*}
Hence, $(\ast)$ was proved.
}

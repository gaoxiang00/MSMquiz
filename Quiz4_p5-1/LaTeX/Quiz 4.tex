% Binomial Distribution

\section{Binomial Distribution}

\frame{
\frametitle{Binomial Distribution}
Let $X\sim Bin(n,p)$. Then:
\begin{equation*}
\begin{align*}
\varphi_X(t)&=\mathbb{E}[e^{itX}]=\sum_{k=0}^n e^{itk}\cdot \mathbb{P}(X=k) \\
				    &=\sum_{k=0}^n e^{itk}\cdot \binom{n}{k}\cdot p^k\cdot (1-p)^{n-k} \\
						&=\sum_{k=0}^n \binom{n}{k}\cdot (e^{it} \cdot p)^k \cdot (1-p)^{n-k} \\
						&= [e^{it} \cdot p + (1-p)]^n
\end{align*}
\end{equation*} \\ \vspace{3mm}
because $(a+b)^n=\sum_0^n \binom{n}{k} a^k b^{n-k}$ for $n\in \mathbb{N}$.

}

% Poisson Distribution

\section{Poisson Distribution}

\frame{
\frametitle{Poisson Distribution}
Let $X\sim Pois(\lambda)$. Then:
\begin{equation*}
\begin{align*}
\varphi_X(t)&=\mathbb{E}[e^{itX}]=\sum_{k=0}^{\infty} e^{itk} \cdot \mathbb{P}(X=k) \\
						&=\sum_{k=0}^{\infty} e^{itk} \cdot e^{-\lambda} \cdot \frac{\lambda^k}{k!} \\
						&= e^{-\lambda}\cdot \sum_{k=0}^{\infty} \frac{(\lambda\cdot e^{it})^k}{k!} \\
						&= e^{-\lambda}\cdot e^{\lambda e^{it}}= e^{\lambda (e^{it}-1)}
\end{align*}
\end{equation*}
because $e^x = \sum_0^\infty \frac{x^k}{k!}$ for $x\in \mathbb{R}$.
}

% Standard Normal Distrbution

\section{Standard Normal Distribution}

\frame{
\frametitle{Standard Normal Distribution}
Let $X\sim N(0,1)$. Then:
\begin{equation*}
\begin{align*}
\varphi_X(t)&=\mathbb{E}[e^{itX}]=\int_{-\infty}^\infty e^{itx} \cdot f(x)\ \mathrm{d}x \\
						&=\int_{-\infty}^\infty e^{itx} \cdot \frac{1}{\sqrt{2\pi}} e^{-\frac{x^2}{2}} \mathrm{d}x \\
						&=\frac{1}{\sqrt{2\pi}} [\int_{-\infty}^0 e^{itx}\cdot e^{-\frac{x^2}{2}}\ \mathrm{d}x + \int_0^\infty e^{itx}\cdot e^{-\frac{x^2}{2}}\ \mathrm{d}x] \\
						&=\frac{1}{\sqrt{2\pi}} [\int_0^\infty e^{-itx}\cdot e^{-\frac{x^2}{2}}\ \mathrm{d}x + \int_0^\infty e^{itx}\cdot e^{-\frac{x^2}{2}}\ \mathrm{d}x]
\end{align*}
\end{equation*}

}

\frame{
\frametitle{Standard Normal Distribution}

\begin{equation*}
\begin{align*}
\varphi_X(t)&=\frac{1}{\sqrt{2\pi}}\cdot\int_0^\infty e^{-\frac{x^2}{2}}\cdot e^{itx} + e^{-\frac{x^2}{2}}\cdot e^{-itx}\ \mathrm{d}x \\
&=\frac{1}{\sqrt{2\pi}}\cdot\int_0^\infty e^{-\frac{x^2}{2}}\cdot [e^{itx} + e^{-itx}]\ \mathrm{d}x \\
&=\frac{1}{\sqrt{2\pi}}\cdot\int_0^\infty e^{-\frac{x^2}{2}}\cdot 2 \cdot cos(tx)\ \mathrm{d}x \\
&=\frac{2}{\sqrt{2\pi}}\cdot\int_0^\infty e^{-\frac{x^2}{2}}\cdot cos(tx)\ \mathrm{d}x \\
&\overset{\text{(1)}}{=}e^{-\frac{t^2}{2}}
\end{align*}
\end{equation*}

}

\frame{
\frametitle{Standard Normal Distribution}
To prove (1) let 
\begin{equation*}
F(t):=\varphi_X(t)=\frac{2}{\sqrt{2\pi}}\cdot \int_0^\infty e^{-\frac{x^2}{2}}\cdot cos(tx)\ \mathrm{d}x
\end{equation*}

We show that 
\begin{equation*}
F'(t)=-t\cdot F(t) \eqno  (\ast)
\end{equation*}
This ordinary differential equation has the solution $F(t)=c\cdot e^{-\frac{t^2}{2}}$, where c is a constant. Since we have 
\begin{equation*}
F(0)=\int_{-\infty}^\infty \frac{1}{\sqrt{2\pi}}\cdot e^{-\frac{x^2}{2}}\ \mathrm{d}x = 1
\end{equation*}
we get $c=1$. Hence, $\varphi_X(t)=F(t)=e^{-\frac{t^2}{2}}$.
}

\frame{
\frametitle{Standard Normal Distribution}

\begin{equation*}
\begin{align*}
F'(t)&=\frac{d}{dt}\ \frac{2}{\sqrt{2\pi}}\cdot \int_0^\infty e^{-\frac{x^2}{2}}\cdot cos(tx) \mathrm{d}x \\
&=\frac{2}{\sqrt{2\pi}}\cdot \int_0^\infty e^{-\frac{x^2}{2}}\cdot \frac{d}{dt} cos(tx) \mathrm{d}x \\
&=\frac{2}{\sqrt{2\pi}}\cdot \int_0^\infty -e^{-\frac{x^2}{2}}\cdot x\cdot sin(tx) \mathrm{d}x
\end{align*}
\end{equation*}

Integration by parts: \\ $u(x)=e^{-\frac{x^2}{2}}$, $u'(x)=-x\cdot e^{-\frac{x^2}{2}}$ and  \\ $v(x)=sin(tx)$, $v'(x)=t\dcot cos(tx)$

}

\frame{
\frametitle{Standard Normal Distribution}

\begin{equation*}
\begin{align*}
F'(t)&=\frac{2}{\sqrt{2\pi}}\cdot [(e^{-\frac{x^2}{2}}\cdot sin(tx))|_0^\infty-\int_0^\infty e^{-\frac{x^2}{2}}\cdot t \cdot cos(tx)\ \mathrm{d}x] \\
&=-\frac{2}{\sqrt{2\pi}}\cdot \int_0^\infty e^{-\frac{x^2}{2}}\cdot t \cdot cos(tx)\ \mathrm{d}x \\
&=-t\cdot \int_0^\infty \frac{1}{\sqrt{2\pi}}e^{-\frac{x^2}{2}}\cdot 2 \cdot cos(tx)\ \mathrm{d}x \\
&=-t\cdot F(t)
\end{align*}
\end{equation*}
Hence, $(\ast)$ was proved and we get $\varphi_X(t)=e^{-\frac{t^2}{2}}$.

}
